```
## JOINS — Combinar tablas

Imagina estas dos tablas:
```
clientes: id_cliente, nombre
pedidos:  id_pedido, id_cliente, producto

-- INNER JOIN: solo los que coinciden en AMBAS tablas
SELECT c.nombre, p.producto
FROM clientes c
INNER JOIN pedidos p ON c.id_cliente = p.id_cliente;

-- LEFT JOIN: todos los de la izquierda, coincidan o no
-- (clientes aunque no tengan pedidos)
SELECT c.nombre, p.producto
FROM clientes c
LEFT JOIN pedidos p ON c.id_cliente = p.id_cliente;

-- RIGHT JOIN: todos los de la derecha, coincidan o no
-- (pedidos aunque no tengan cliente asociado)
SELECT c.nombre, p.producto
FROM clientes c
RIGHT JOIN pedidos p ON c.id_cliente = p.id_cliente;

-- FULL JOIN: todos de ambas tablas, coincidan o no
SELECT c.nombre, p.producto
FROM clientes c
FULL OUTER JOIN pedidos p ON c.id_cliente = p.id_cliente;

Visualización de JOINs
JOIN ¿Qué trae? INNER Solo los que coinciden en ambas tablas
LEFT Todos de la izquierda + coincidencias
RIGHT Todos de la derecha + coincidencias
FULL Todos de ambas tablas

JOIN con 3 tablas.
SELECT c.nombre, p.producto, cat.nombre AS categoria
FROM clientes c
INNER JOIN pedidos p ON c.id_cliente = p.id_cliente
INNER JOIN categorias cat ON p.id_categoria = cat.id_categoria;

PROCEDIMIENTOS ALMACENADOS:
-- Crear
CREATE OR REPLACE FUNCTION obtener_por_categoria(cat VARCHAR)
RETURNS TABLE(id INT, nombre VARCHAR, precio NUMERIC) AS $$
BEGIN
  RETURN QUERY 
  SELECT p.id, p.nombre, p.precio 
  FROM productos p 
  WHERE p.categoria = cat;
END;
$$ LANGUAGE plpgsql;

-- Ejecutar
SELECT * FROM obtener_por_categoria('Ropa');

-- Eliminar
DROP FUNCTION obtener_por_categoria;
